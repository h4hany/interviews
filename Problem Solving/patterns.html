Sliding Window نافذة منزلقة

بتمشي على array أو string بنافذة بتكبر وبتصغر

💡 الفكرة بمثال من الحياة

تخيل إنك شايل مظروف وبتمشي على صف من الناس وبتاخد فلوس منهم. المظروف بيتسع لـ 3 ناس بس. لما تاخد واحد جديد من الأمام، الواحد اللي وراك بيطلع من المظروف. أنت مش بتحسب الكل من أول كل مرة — بس بتضيف الجديد وبتشيل القديم! ده هو الـ Sliding Window. بدل ما تجرب كل الاحتمالات (O(n²)) بتمشي مرة واحدة (O(n)).

⏰ إمتى تستخدمه؟

    ✅ استخدمه لو
  • بتدور على subarray/substring
  • فيه شرط على نطاق متتالي
  • الكلمات: "أطول / أقصر / أكبر مجموع"
  • البيانات متتالية (array / string)
    ❌ متستخدمهش لو
  • العناصر مش متتالية
  • محتاج تقارن بعيدين من بعض
  • البيانات مش مرتبة بشكل معين

🪜 الخطوات

1

حدد طرفين: left و right يبدأوا من 0

2

وسّع من اليمين: right++ وأضيف العنصر الجديد

3

لو الشرط اتكسر: ضيّق من الشمال left++

4

حدّث الجواب: في كل خطوة احسب الأفضل

👁️ مثال مرئي - Longest Substring Without Repeating

الكلمة:  a  b  c  a  b  c  b  b
Index:   0  1  2  3  4  5  6  7

خطوة 1:  [a]              ← أضيف a ✓  أطول = 1
خطوة 2:  [a  b]           ← أضيف b ✓  أطول = 2
خطوة 3:  [a  b  c]        ← أضيف c ✓  أطول = 3
خطوة 4:  [a  b  c  a] ← a موجود! ضيّق
          Left يتحرك →
          [b  c  a]        ← شيل a الأول أطول = 3
خطوة 5:  [b  c  a  b] ← b موجود! ضيّق
          [c  a  b]        ← شيل b الأول أطول = 3
الجواب النهائي: 3 ✅

💻 الكود مع شرح كل سطر

slidingWindow.js 
// 🎯 Sliding Window — أطول substring من غير تكرار
function lengthOfLongestSubstring(s) {

  // Set بيحفظ الحروف اللي في النافذة الحالية
  const seen = new Set();

  let left = 0;   // ← الطرف الأيسر (بداية النافذة)
  let maxLen = 0; // ← أطول نافذة لاقيناها

  // right بيمشي من الأول للآخر (يمثل آخر النافذة)
  for (let right = 0; right < s.length; right++) {

    const char = s[right]; // الحرف الجديد اللي دخل النافذة

    // لو الحرف ده موجود قبل كده في النافذة...
    while (seen.has(char)) {
      // شيل الحرف من الطرف الأيسر واحد واحد
      seen.delete(s[left]);
      left++; // ضيّق النافذة من الشمال
    }

    // أضيف الحرف الجديد للنافذة
    seen.add(char);

    // right - left + 1 = حجم النافذة الحالية
    maxLen = Math.max(maxLen, right - left + 1);
  }

  return maxLen; // ← الجواب النهائي
}

// ✅ اختبار: "abcabcbb" → 3
console.log(lengthOfLongestSubstring("abcabcbb")); // 3
⏱ Time: O(n) 💾 Space: O(min(n, alphabet))

Two Pointers إصبعين

إصبعين بيتحركوا في نفس الـ array من اتجاهات مختلفة

💡 الفكرة بمثال من الحياة

تخيل كتاب وعايز تعرف هل بيتقرأ نفسه من الاتجاهين (palindrome). بتحط إصبع في الصفحة الأولى وإصبع في الصفحة الأخيرة. لو متطابقين، الاتنين بيتحركوا للوسط. لو لأ — مش palindrome. ده بدل ما تقلب كل الصفحات وتقارن. نوعين: (1) واحد من كل طرف يتقابلوا في الوسط. (2) واحد سريع وواحد بطيء في نفس الاتجاه.

⏰ إمتى تستخدمه؟

    ✅ استخدمه لو
  • Array مرتبة وبتدور على زوجين
  • بتدور على palindrome
  • بتدور على cycle في linked list
  • بتحتاج تقليل O(n²) لـ O(n)
    ❌ متستخدمهش لو
  • Array مش مرتبة (في بعض المسائل)
  • محتاج تتذكر أكتر من موقعين

💻 الكود - Container With Most Water

twoPointers.js
// 🎯 أكبر حوض مياه — Two Pointers
function maxArea(height) {

  let left = 0;                      // إصبع في أول حيط
  let right = height.length - 1;  // إصبع في آخر حيط
  let max = 0;                       // أكبر مساحة لقيناها

  // لحد ما الإصبعين يلتقوا في الوسط
  while (left < right) {

    // عرض الحوض = المسافة بين الحيطين
    const width = right - left;

    // ارتفاع الحوض = أقصر الحيطين (المياه بتطفح من الأقصر)
    const h = Math.min(height[left], height[right]);

    // احسب المساحة وحدّث الـ max
    max = Math.max(max, width * h);

    // حرّك الإصبع اللي جنب الحيط الأقصر
    // (الأطول مهوش المشكلة، المشكلة في الأقصر)
    if (height[left] < height[right]) {
      left++;   // الحيط الأيسر أقصر → حرّكه
    } else {
      right--;  // الحيط الأيمن أقصر → حرّكه
    }
  }

  return max;
}

// ✅ اختبار: [1,8,6,2,5,4,8,3,7] → 49
console.log(maxArea([1,8,6,2,5,4,8,3,7])); // 49
⏱ Time: O(n) 💾 Space: O(1)

Dynamic Programming برمجة ديناميكية

احفظ نتائج المسائل الصغيرة عشان ما تحلهاش تاني

💡 الفكرة بمثال من الحياة

تخيل بتحسب 5! (factorial). لو حسبت 4! من قبل، ما تحسبهاش تاني — بس اضرب 5 × 4!. DP = "أنا مش بحل نفس المسألة أكتر من مرة." بتقسم المسألة الكبيرة لمسائل أصغر، بتحل الصغيرة وبتحفظ نتيجتها في جدول (Memoization أو Tabulation)، وبتبني الكبيرة على الصغيرة. الأسئلة: "كام طريقة؟" أو "أقل / أكبر / أمثل" = DP في الغالب!

🔑 نوعين DP

1

Top-Down (Memoization): ابدأ من المسألة الكبيرة ونزّل للصغيرة. استخدم Recursion + Cache.

2

Bottom-Up (Tabulation): ابدأ من أصغر مسألة وابني لفوق. استخدم Array/Table.

💻 الكود - Climbing Stairs + Coin Change

dynamicProgramming.js
// ══════════════════════════════════════
// 🎯 مثال 1: Climbing Stairs (Bottom-Up)
// ══════════════════════════════════════
function climbStairs(n) {

  // لو درجة 1 أو 2 — الجواب واضح
  if (n <= 2) return n;

  // dp[i] = كام طريقة توصل للدرجة i
  const dp = [0, 1, 2]; // Base cases

  for (let i = 3; i <= n; i++) {
    // طرق الوصول للدرجة i = طرق i-1 + طرق i-2
    // (جيت من درجة قبلها بخطوة) + (جيت من درجتين قبلها بقفزة)
    dp[i] = dp[i-1] + dp[i-2];
  }

  return dp[n]; // الجواب
}

// ══════════════════════════════════════
// 🎯 مثال 2: Coin Change (Bottom-Up)
// ══════════════════════════════════════
function coinChange(coins, amount) {

  // dp[i] = أقل عدد عملات لصرف مبلغ i
  // بنملاه بـ Infinity = "مستحيل" في البداية
  const dp = new Array(amount + 1).fill(Infinity);

  // مبلغ 0 محتاج 0 عملات — Base case
  dp[0] = 0;

  // لكل مبلغ من 1 للـ amount
  for (let i = 1; i <= amount; i++) {
    // جرب كل عملة
    for (const coin of coins) {

      // لو العملة أصغر من أو تساوي المبلغ
      if (coin <= i) {
        // dp[i-coin] = أقل عملات للباقي
        // +1 للعملة الحالية
        dp[i] = Math.min(dp[i], dp[i-coin] + 1);
      }
    }
  }

  // لو فضل Infinity → مستحيل نصرف المبلغ ده
  return dp[amount] === Infinity ? -1 : dp[amount];
}

console.log(climbStairs(5));                    // 8
console.log(coinChange([1,5,6,9], 11));     // 2
⏱ Time: O(n) / O(n×coins) 💾 Space: O(n)

BFS — Breadth First Search البحث بالعرض

بتكشف المستوى الأول، بعدين الثاني، بعدين الثالث...

💡 الفكرة بمثال من الحياة

تخيل بتدور على صاحبك في مدينة. بتبعت رسائل لكل الناس اللي تعرفهم. كل واحد منهم بيبعت لكل اللي يعرفهم. بتكشف دايرة دايرة للخارج. BFS بتستخدم Queue (طابور) — الأقرب أول. مهم جداً لأقصر مسار في graph بدون أوزان.

💻 الكود - Level Order Tree Traversal

bfs.js
// 🎯 BFS على شجرة — قراءة مستوى مستوى
function levelOrder(root) {
  // لو الشجرة فاضية — ارجع array فاضي
  if (!root) return [];

  const result = [];          // النتيجة النهائية
  const queue = [root];      // الطابور — ابدأ بالجذر

  // لحد ما الطابور يفضى
  while (queue.length > 0) {

    // عدد العقد في المستوى الحالي
    const levelSize = queue.length;
    const level = [];           // قيم المستوى ده

    // اشتغل على كل عقدة في المستوى الحالي
    for (let i = 0; i < levelSize; i++) {

      // اطلع أول عقدة من الطابور
      const node = queue.shift();

      // حط قيمتها في مصفوفة المستوى
      level.push(node.val);

      // لو فيه ابن أيسر، حطه في الطابور للمستوى الجاي
      if (node.left)  queue.push(node.left);

      // لو فيه ابن أيمن، حطه في الطابور للمستوى الجاي
      if (node.right) queue.push(node.right);
    }

    // ضيف المستوى كاملاً للنتيجة
    result.push(level);
  }

  return result;
}
// ✅ الشجرة [3,9,20,null,null,15,7] → [[3],[9,20],[15,7]]
⏱ Time: O(n) 💾 Space: O(w) — w = عرض الشجرة

DFS — Depth First Search البحث بالعمق

روح لأعمق نقطة الأول، وارجع، وجرب الطريق التاني

💡 الفكرة بمثال من الحياة

تخيل بتتفرج على متاهة. بتدخل في أول طريق وبتمشي لأخره. لو وصلت لنهاية مسدودة، ترجع خطوة وتجرب الطريق التاني. بتستخدم Stack (أو Recursion). مهم لـ: الأشجار، الجرافات، Backtracking، إيجاد كل المسارات الممكنة.

💻 الكود - Number of Islands (DFS على Matrix)

dfs.js
// 🎯 عد الجزر — DFS على مصفوفة
function numIslands(grid) {

  // لو ما فيش grid → صفر جزر
  if (!grid || grid.length === 0) return 0;

  let count = 0; // عداد الجزر

  // ابحث في كل خلية
  for (let r = 0; r < grid.length; r++) {
    for (let c = 0; c < grid[0].length; c++) {

      // لو لاقيت '1' (أرض) → جزيرة جديدة!
      if (grid[r][c] === '1') {
        count++;
        dfs(grid, r, c); // احرق الجزيرة كلها عشان ما تعدهاش تاني
      }
    }
  }
  return count;
}

function dfs(grid, r, c) {
  // توقف لو خرجنا من الحدود أو وصلنا لمياه أو زرنا من قبل
  if (r < 0 || r >= grid.length ||
      c < 0 || c >= grid[0].length ||
      grid[r][c] !== '1') return;

  // "احرق" الخلية (غيّرها لـ '0') = زيارة
  grid[r][c] = '0';

  // زور الأربع اتجاهات
  dfs(grid, r+1, c); // تحت
  dfs(grid, r-1, c); // فوق
  dfs(grid, r, c+1); // يمين
  dfs(grid, r, c-1); // شمال
}
⏱ Time: O(n × m) 💾 Space: O(n × m) stack

Binary Search البحث الثنائي

اقسم المشكلة لنصين واحذف نص كل مرة

💡 الفكرة بمثال من الحياة

بتدور على اسم في قاموس ورقي. بتفتحه من النص. لو الاسم قبل الصفحة دي — روح النص الأول. لو بعدها — النص الثاني. كل مرة بتشيل نص الكتاب! ده بيحوّل O(n) لـ O(log n). شرط واحد مهم: البيانات لازم تكون مرتبة.

💻 الكود - Classic Binary Search Template

binarySearch.js
// 🎯 البحث الثنائي — Template ينفع لمعظم المسائل
function binarySearch(nums, target) {

  let left  = 0;                  // بداية نطاق البحث
  let right = nums.length - 1;  // نهاية نطاق البحث

  while (left <= right) {

    // النص — مش (left + right) / 2 عشان نتجنب overflow
    const mid = Math.floor((left + right) / 2);

    if (nums[mid] === target) {
      return mid;     // ✅ لاقيناه!
    } else if (nums[mid] < target) {
      left = mid + 1;  // الـ target في النص الأيمن
    } else {
      right = mid - 1; // الـ target في النص الأيسر
    }
  }

  return -1; // ما لقيناهوش
}

console.log(binarySearch([1,3,5,7,9], 7)); // 3 (index)
⏱ Time: O(log n) 💾 Space: O(1)

Stack & Monotonic Stack الكومة

آخر حاجة دخلت أول حاجة بتطلع (LIFO)

💡 الفكرة بمثال من الحياة

كومة طباق. آخر طبق حطيته فوق = أول طبق بتاخده. Stack عادي للأقواس والتعبيرات الحسابية. Monotonic Stack خاص: الكومة دايماً مرتبة (صاعدة أو نازلة). مفيد لـ "إيجاد أول عنصر أكبر/أصغر على اليمين أو اليسار".

💻 الكود - Valid Parentheses + Monotonic Stack

stack.js
// ══════════════════════════
// 🎯 Valid Parentheses
// ══════════════════════════
function isValid(s) {
  const stack = [];  // الكومة — بتحفظ الأقواس المفتوحة
  const map = { ')': '(', '}': '{', ']': '[' };

  for (const ch of s) {
    if ('({['.includes(ch)) {
      stack.push(ch);       // قوس مفتوح → حطه في الكومة
    } else {
      // قوس مغلق → لازم يطابق آخر قوس في الكومة
      if (stack.pop() !== map[ch]) return false;
    }
  }

  // الكومة لازم تبقى فاضية في الآخر
  return stack.length === 0;
}

// ══════════════════════════════════
// 🎯 Monotonic Stack — Largest Rectangle
// ══════════════════════════════════
function largestRectangle(heights) {
  const stack = [];  // بتحفظ indices
  let maxArea = 0;

  // أضيف 0 في الآخر عشان نفرّغ الـ stack
  heights.push(0);

  for (let i = 0; i < heights.length; i++) {

    // لما ارتفاع جديد أقل من رأس الـ stack
    // ← ده معناه الشريط اللي فوق لقى حده
    while (stack.length && heights[i] < heights[stack[stack.length-1]]) {

      const h = heights[stack.pop()];  // ارتفاع الشريط
      const w = stack.length ? i - stack[stack.length-1] - 1 : i; // العرض

      maxArea = Math.max(maxArea, h * w);
    }
    stack.push(i); // حط الـ index الحالي
  }
  return maxArea;
}

Backtracking الرجوع للخلف

جرب كل الاحتمالات وارجع لو ما نفعتش

💡 الفكرة بمثال من الحياة

تخيل بتحل كلمة متقاطعة. بتكتب حرف، لو ما نفعش امسحه وجرب التاني. Backtracking = "جرب، لو غلط ارجع خطوة وجرب غيره." بتبني الحل تدريجياً وبتشيل الاختيار الغلط. Pattern: اختار → ابنى → Recurse → امسح (Undo).

💻 الكود - Generate All Subsets

backtracking.js
// 🎯 Backtracking Template — جنّر كل الـ Subsets
function subsets(nums) {
  const result = [];  // كل النتائج

  function backtrack(start, current) {
    // ✅ حط الحل الحالي في النتائج
    result.push([...current]);

    // جرب كل عنصر من الـ start للآخر
    for (let i = start; i < nums.length; i++) {

      // ① اختار: أضيف العنصر
      current.push(nums[i]);

      // ② ابنى: نزّل أكتر مع العنصر الجاي
      backtrack(i + 1, current);

      // ③ امسح (Undo): شيل العنصر اللي أضفناه
      // ← ده جوهر الـ Backtracking!
      current.pop();
    }
  }

  backtrack(0, []);
  return result;
}

// ✅ [1,2,3] → [[],[1],[1,2],[1,2,3],[1,3],[2],[2,3],[3]]
⏱ Time: O(2ⁿ) 💾 Space: O(n)

Greedy الجشع

خد أفضل قرار دلوقتي بدون ما تفكر في المستقبل

💡 الفكرة بمثال من الحياة

تخيل عندك فكة (عملات) وعايز تدي زبون باقي بأقل عملات. بتاخد دايماً أكبر عملة ممكنة الأول. 87 جنيه: خد 50 (فضل 37) → خد 25 (فضل 12) → خد 10 (فضل 2) → خد 2. ده Greedy — "أحسن قرار محلي يوصل لأحسن قرار كلي". مش دايماً شغّال — لازم تثبت إن الـ Greedy choice صح!

💻 الكود - Best Time to Buy & Sell Stock

greedy.js
// 🎯 Greedy — أفضل وقت للشراء والبيع
function maxProfit(prices) {

  // بنفضل نتذكر أرخص سعر شفناه لحد دلوقتي
  let minPrice = Infinity;
  let maxProfit = 0;

  for (const price of prices) {

    // لو الأسعار النهارده أرخص من أرخص سعر شفناه
    if (price < minPrice) {
      minPrice = price; // ← ده أحسن يوم للشراء
    } else {
      // لو بعت النهارده — كام ربحنا؟
      maxProfit = Math.max(maxProfit, price - minPrice);
    }
  }

  return maxProfit;
}

// ✅ [7,1,5,3,6,4] → 5 (اشتري بـ1، بيع بـ6)
console.log(maxProfit([7,1,5,3,6,4])); // 5
⏱ Time: O(n) 💾 Space: O(1)

Topological Sort الترتيب التبولوجي

رتّب المهام بحيث كل مهمة تيجي بعد اللي بتعتمد عليها

💡 الفكرة بمثال من الحياة

عندك وصفة أكل. لازم تقطع الخضار قبل ما تحطها في الطاجن، والطاجن لازم يتسخن قبل ما تأكل. Topological Sort بيرتب المهام بحيث كل مهمة تيجي بعد كل اللي بتعتمد عليه. لو فيه دورة (A يحتاج B وB يحتاج A) — مستحيل تكمل!

💻 الكود - Course Schedule (Kahn's Algorithm)

topologicalSort.js
// 🎯 Topological Sort — Kahn's BFS Algorithm
function canFinish(numCourses, prerequisites) {

  // inDegree[i] = كام كورس محتاج تخلص قبل كورس i
  const inDegree = new Array(numCourses).fill(0);

  // graph[i] = الكورسات اللي بتعتمد على كورس i
  const graph = new Array(numCourses).fill(null).map(() => []);

  // ابنى الجراف واحسب الـ inDegrees
  for (const [course, prereq] of prerequisites) {
    graph[prereq].push(course);  // prereq → course
    inDegree[course]++;           // course محتاج prereq
  }

  // ابدأ بالكورسات اللي ما ليهاش prerequisites
  const queue = [];
  for (let i = 0; i < numCourses; i++) {
    if (inDegree[i] === 0) queue.push(i);
  }

  let completed = 0; // عداد الكورسات اللي خلصناها

  while (queue.length) {
    const course = queue.shift(); // خد كورس من الطابور
    completed++;                   // خلصناه!

    // لكل كورس بيعتمد على الكورس ده
    for (const next of graph[course]) {
      inDegree[next]--;                     // شيل prerequisite
      if (inDegree[next] === 0) queue.push(next); // أصبح جاهز!
    }
  }

  // لو خلصنا كل الكورسات → مفيش دورة → ممكن
  return completed === numCourses;
}

console.log(canFinish(2, [[1,0]]));         // true
console.log(canFinish(2, [[1,0],[0,1]])); // false (حلقة!)
⏱ Time: O(V + E) 💾 Space: O(V + E)
